  // Función que genera el GIF desde listaImagenes
  function createGifFromList() {
    const gif = new GIF({
    workers: 2,
    quality: 10
    });
    listaImagenes.forEach((blob) => {
    gif.addFrame(blob, { delay: 100 });
    });
    gif.on('finished', (blob) => {
    const url = URL.createObjectURL(blob);
    const img = document.createElement('img');
    img.src = url;
   // document.getElementById('framesContainer').appendChild(img[2]);
    });
    gif.render();
    }
